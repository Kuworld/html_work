import matplotlib.pyplot as plt import numpy as np # 얼굴 그리기 def draw_face(): circle = plt.Circle((0.5, 0.5), 0.4, color='lightblue') plt.gca().add_patch(circle) # 눈 그리기 def draw_eyes(): left_eye = plt.Circle((0.4, 0.6), 0.05, color='black') right_eye = plt.Circle((0.6, 0.6), 0.05, color='black') plt.gca().add_patch(left_eye) plt.gca().add_patch(right_eye) # 코 그리기 def draw_nose(): nose = plt.Polygon([[0.5, 0.55], [0.48, 0.45], [0.52, 0.45]], color='black') plt.gca().add_patch(nose) # 입 그리기 def draw_mouth(): mouth = plt.Polygon([[0.45, 0.4], [0.55, 0.4], [0.5, 0.35]], color='black') plt.gca().add_patch(mouth) # 그림판 설정 plt.figure() plt.axis('scaled') plt.axis('off') # 얼굴 그리기 draw_face() # 눈 그리기 draw_eyes() # 코 그리기 draw_nose() # 입 그리기 draw_mouth() # 그림 출력 plt.show()
